public static void BankAlgorithm(Customer c){
        Scanner sc = new Scanner(System.in);
        String s;
        outer_loop: while (true) {

            switch (num) {
                case 1:
                    System.out.println("Enter the Amount (How much amount you want to deposit)");
                    float n = Integer.parseInt(sc.nextLine());
                    c.deposit(n);
                    System.out.println("Do you want to continue");
                    s = sc.nextLine();
                    if (s.equals("yes")) {
                        Bank.menuBar();
                        // System.out.println("Enter Your Choice");
                        // num = Integer.parseInt(sc.nextLine());
                        break;
                    } else {
                        c.login();
                        break outer_loop;
                    }

                case 2:
                    System.out.println("Enter the amount you want to withdraw");
                    float m = Integer.parseInt(sc.nextLine());
                    c.withDrawal(m);
                    System.out.println("Do you want to continue");
                    s = sc.nextLine();
                    if (s.equals("yes")) {
                        Bank.menuBar();
                        // System.out.println("Enter Your Choice");
                        // num = Integer.parseInt(sc.nextLine());
                        break;
                    } else {
                        c.login();
                        break outer_loop;
                    }

                case 3:
                    c.checkBalance();

                        Bank.menuBar();
                        // System.out.println("Enter Your Choice");
                        // num = Integer.parseInt(sc.nextLine());
                        break;
                    } else {
                        c.login();
                        break outer_loop;
                    }

                case 4:
                    c.editProfile();
                    System.out.println("Do you want to continue");
                    s = sc.nextLine();

                        Bank.menuBar();
                        // System.out.println("Enter Your Choice");
                        // num = Integer.parseInt(sc.nextLine());
                        break;

                        c.login();
                        break outer_loop;
                    }
                case 5:
                    c.changePassword();
                    System.out.println("Do you want to continue");
                    s = sc.nextLine();
                    if (s.equals("yes")) {
                        Bank.menuBar();
                        // System.out.println("Enter Your Choice");
                        // num = Integer.parseInt(sc.nextLine());
                        break;

                        c.login();
                        break outer_loop;

                case 6:

        }
